83void PublisherInterface::publish() {
84 try {
85 if (this->message_pair_ == nullptr) {
86 throw exceptions::NullPointerException("Message pair is not set, nothing to publish");
87 }
88 switch (this->message_pair_->get_type()) {
89 case MessageType::BOOL:
90 this->publish(this->message_pair_->write<std_msgs::msg::Bool, bool>());
91 break;
92 case MessageType::FLOAT64:
93 this->publish(this->message_pair_->write<std_msgs::msg::Float64, double>());
94 break;
95 case MessageType::FLOAT64_MULTI_ARRAY:
96 this->publish(this->message_pair_->write<std_msgs::msg::Float64MultiArray, std::vector<double>>());
97 break;
98 case MessageType::INT32:
99 this->publish(this->message_pair_->write<std_msgs::msg::Int32, int>());
100 break;
101 case MessageType::STRING:
102 this->publish(this->message_pair_->write<std_msgs::msg::String, std::string>());
103 break;
104 case MessageType::ENCODED_STATE:
105 this->publish(this->message_pair_->write<EncodedState, state_representation::State>());
106 break;
107 }
108 } catch (const exceptions::CoreException& ex) {
109 throw;
110 }
