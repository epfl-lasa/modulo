265  && this->get_current_state().id() != lifecycle_msgs::msg::State::PRIMARY_STATE_INACTIVE) {
266  RCLCPP_WARN_STREAM_THROTTLE(this->get_logger(), *this->get_clock(), 1000,
267  "Adding output in state " << this->get_current_state().label() << " is not allowed.");
268  return;
269  }
270  try {

272  } catch (const std::exception& ex) {

275  }
