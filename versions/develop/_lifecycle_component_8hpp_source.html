276inline void LifecycleComponent::add_output(
277 const std::string& signal_name, const std::shared_ptr<DataT>& data, const std::string& default_topic,
278 bool fixed_topic
279) {
280 if (this->get_current_state().id() != lifecycle_msgs::msg::State::PRIMARY_STATE_UNCONFIGURED
281 && this->get_current_state().id() != lifecycle_msgs::msg::State::PRIMARY_STATE_INACTIVE) {
282 RCLCPP_WARN_STREAM_THROTTLE(this->get_logger(), *this->get_clock(), 1000,
283 "Adding output in state " << this->get_current_state().label() << " is not allowed.");
284 return;
285 }
286 try {
287 this->create_output(signal_name, data, default_topic, fixed_topic);
288 } catch (const exceptions::AddSignalException& ex) {
289 RCLCPP_ERROR_STREAM(this->get_logger(), "Failed to add output '" << signal_name << "': " << ex.what());
290 }
291}
