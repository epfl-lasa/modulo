260inline void LifecycleComponent::add_output(
261 const std::string& signal_name, const std::shared_ptr<DataT>& data, const std::string& default_topic,
262 bool fixed_topic
263) {
264 if (this->get_current_state().id() != lifecycle_msgs::msg::State::PRIMARY_STATE_UNCONFIGURED
265 && this->get_current_state().id() != lifecycle_msgs::msg::State::PRIMARY_STATE_INACTIVE) {
266 RCLCPP_WARN_STREAM_THROTTLE(this->get_logger(), *this->get_clock(), 1000,
267 "Adding output in state " << this->get_current_state().label() << " is not allowed.");
268 return;
269 }
270 try {
271 this->create_output(signal_name, data, default_topic, fixed_topic);
272 } catch (const exceptions::AddSignalException& ex) {
273 RCLCPP_ERROR_STREAM(this->get_logger(), "Failed to add output '" << signal_name << "': " << ex.what());
274 }
275}
