50  RCLCPP_DEBUG(this->get_logger(), "on_configure called from previous state %s", previous_state.label().c_str());
51  if (previous_state.id() != lifecycle_msgs::msg::State::PRIMARY_STATE_UNCONFIGURED) {
52  RCLCPP_WARN(get_logger(), "Invalid transition 'configure' from state %s.", previous_state.label().c_str());
53  return rclcpp_lifecycle::node_interfaces::LifecycleNodeInterface::CallbackReturn::FAILURE;
54  }
55  if (!this->handle_configure()) {
56  RCLCPP_WARN(get_logger(), "Configuration failed! Reverting to the unconfigured state.");
57  // perform cleanup actions to ensure the component is unconfigured
58  if (this->handle_cleanup()) {
59  return rclcpp_lifecycle::node_interfaces::LifecycleNodeInterface::CallbackReturn::FAILURE;
60  } else {
61  RCLCPP_ERROR(get_logger(),
62  "Could not revert to the unconfigured state! Entering into the error processing transition state.");
63  return rclcpp_lifecycle::node_interfaces::LifecycleNodeInterface::CallbackReturn::ERROR;
64  }
65  }
66  return rclcpp_lifecycle::node_interfaces::LifecycleNodeInterface::CallbackReturn::SUCCESS;
67 }

155 LifecycleComponent::on_shutdown(const rclcpp_lifecycle::State& previous_state) {
156  RCLCPP_DEBUG(this->get_logger(), "on_shutdown called from previous state %s", previous_state.label().c_str());
157  switch (previous_state.id()) {
158  case lifecycle_msgs::msg::State::PRIMARY_STATE_FINALIZED:
159  return rclcpp_lifecycle::node_interfaces::LifecycleNodeInterface::CallbackReturn::SUCCESS;
160  case lifecycle_msgs::msg::State::PRIMARY_STATE_ACTIVE:
161  if (!this->handle_deactivate()) {
162  RCLCPP_DEBUG(get_logger(), "Shutdown failed during intermediate deactivation!");
163  break;
164  }
165  [[fallthrough]];
166  case lifecycle_msgs::msg::State::PRIMARY_STATE_INACTIVE:
167  if (!this->handle_cleanup()) {
168  RCLCPP_DEBUG(get_logger(), "Shutdown failed during intermediate cleanup!");
169  break;
170  }
171  [[fallthrough]];
172  case lifecycle_msgs::msg::State::PRIMARY_STATE_UNCONFIGURED:
173  if (!this->handle_shutdown()) {
174  break;
175  }
176  // TODO: reset and finalize all needed properties
177  // this->handlers_.clear();
178  // this->daemons_.clear();
179  // this->parameters_.clear();
180  // this->shutdown_ = true;
181  return rclcpp_lifecycle::node_interfaces::LifecycleNodeInterface::CallbackReturn::SUCCESS;
182  default:
183  RCLCPP_WARN(get_logger(), "Invalid transition 'shutdown' from state %s.", previous_state.label().c_str());
184  break;
185  }
186  RCLCPP_ERROR(get_logger(), "Entering into the error processing transition state.");
187  return rclcpp_lifecycle::node_interfaces::LifecycleNodeInterface::CallbackReturn::ERROR;
