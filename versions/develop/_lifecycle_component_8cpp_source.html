50rclcpp_lifecycle::node_interfaces::LifecycleNodeInterface::CallbackReturn
51LifecycleComponent::on_configure(const rclcpp_lifecycle::State& previous_state) {
52 RCLCPP_DEBUG(this->get_logger(), "on_configure called from previous state %s", previous_state.label().c_str());
53 if (previous_state.id() != lifecycle_msgs::msg::State::PRIMARY_STATE_UNCONFIGURED) {
54 RCLCPP_WARN(get_logger(), "Invalid transition 'configure' from state %s.", previous_state.label().c_str());
55 return rclcpp_lifecycle::node_interfaces::LifecycleNodeInterface::CallbackReturn::FAILURE;
56 }
57 if (!this->handle_configure()) {
58 RCLCPP_WARN(get_logger(), "Configuration failed! Reverting to the unconfigured state.");
59 // perform cleanup actions to ensure the component is unconfigured
60 if (this->handle_cleanup()) {
61 return rclcpp_lifecycle::node_interfaces::LifecycleNodeInterface::CallbackReturn::FAILURE;
62 } else {
63 RCLCPP_ERROR(get_logger(),
64 "Could not revert to the unconfigured state! Entering into the error processing transition state.");
65 return rclcpp_lifecycle::node_interfaces::LifecycleNodeInterface::CallbackReturn::ERROR;
66 }
67 }
68 return rclcpp_lifecycle::node_interfaces::LifecycleNodeInterface::CallbackReturn::SUCCESS;
69}

156rclcpp_lifecycle::node_interfaces::LifecycleNodeInterface::CallbackReturn
157LifecycleComponent::on_shutdown(const rclcpp_lifecycle::State& previous_state) {
158 RCLCPP_DEBUG(this->get_logger(), "on_shutdown called from previous state %s", previous_state.label().c_str());
159 switch (previous_state.id()) {
160 case lifecycle_msgs::msg::State::PRIMARY_STATE_FINALIZED:
161 return rclcpp_lifecycle::node_interfaces::LifecycleNodeInterface::CallbackReturn::SUCCESS;
162 case lifecycle_msgs::msg::State::PRIMARY_STATE_ACTIVE:
163 if (!this->handle_deactivate()) {
164 RCLCPP_DEBUG(get_logger(), "Shutdown failed during intermediate deactivation!");
165 break;
166 }
167 [[fallthrough]];
168 case lifecycle_msgs::msg::State::PRIMARY_STATE_INACTIVE:
169 if (!this->handle_cleanup()) {
170 RCLCPP_DEBUG(get_logger(), "Shutdown failed during intermediate cleanup!");
171 break;
172 }
173 [[fallthrough]];
174 case lifecycle_msgs::msg::State::PRIMARY_STATE_UNCONFIGURED:
175 if (!this->handle_shutdown()) {
176 break;
177 }
178 // TODO: reset and finalize all needed properties
179 // this->handlers_.clear();
180 // this->daemons_.clear();
181 // this->parameters_.clear();
182 // this->shutdown_ = true;
183 return rclcpp_lifecycle::node_interfaces::LifecycleNodeInterface::CallbackReturn::SUCCESS;
184 default:
185 RCLCPP_WARN(get_logger(), "Invalid transition 'shutdown' from state %s.", previous_state.label().c_str());
186 break;
187 }
188 RCLCPP_ERROR(get_logger(), "Entering into the error processing transition state.");
189 return rclcpp_lifecycle::node_interfaces::LifecycleNodeInterface::CallbackReturn::ERROR;
190}

228bool LifecycleComponent::configure_outputs() {
229 bool success = true;
230 for (auto& [name, interface]: this->outputs_) {
231 try {
232 auto topic_name = this->get_parameter_value<std::string>(name + "_topic");
233 RCLCPP_DEBUG_STREAM(this->get_logger(),
234 "Configuring output '" << name << "' with topic name '" << topic_name << "'.");
235 auto message_pair = interface->get_message_pair();
236 switch (message_pair->get_type()) {
237 case MessageType::BOOL: {
238 auto publisher = this->create_publisher<std_msgs::msg::Bool>(topic_name, this->qos_);
239 interface = std::make_shared<PublisherHandler<rclcpp_lifecycle::LifecyclePublisher<std_msgs::msg::Bool>,
240 std_msgs::msg::Bool>>(
241 PublisherType::LIFECYCLE_PUBLISHER, publisher
242 )->create_publisher_interface(message_pair);
243 break;
244 }
245 case MessageType::FLOAT64: {
246 auto publisher = this->create_publisher<std_msgs::msg::Float64>(topic_name, this->qos_);

248 std_msgs::msg::Float64>>(
