474 template<class NodeT>
475 inline void ComponentInterface<NodeT>::add_parameter(
476  const std::shared_ptr<state_representation::ParameterInterface>& parameter, const std::string& description,
477  bool read_only
478 ) {
479  this->set_parameter_callback_called_ = false;
480  rclcpp::Parameter ros_param;
481  try {
482  ros_param = modulo_core::translators::write_parameter(parameter);
483  } catch (const modulo_core::exceptions::ParameterTranslationException& ex) {
484  throw exceptions::ComponentParameterException("Failed to add parameter: " + std::string(ex.what()));
485  }
486  if (!NodeT::has_parameter(parameter->get_name())) {
487  RCLCPP_DEBUG_STREAM(this->get_logger(), "Adding parameter '" << parameter->get_name() << "'.");
488  this->parameter_map_.set_parameter(parameter);
489  try {
490  rcl_interfaces::msg::ParameterDescriptor descriptor;
491  descriptor.description = description;
492  descriptor.read_only = read_only;
493  if (parameter->is_empty()) {
494  descriptor.dynamic_typing = true;
495  descriptor.type = modulo_core::translators::get_ros_parameter_type(parameter->get_parameter_type());
496  NodeT::declare_parameter(parameter->get_name(), rclcpp::ParameterValue{}, descriptor);
497  } else {
498  NodeT::declare_parameter(parameter->get_name(), ros_param.get_parameter_value(), descriptor);
499  }
500  if (!this->set_parameter_callback_called_) {
501  auto result = this->on_set_parameters_callback({NodeT::get_parameters({parameter->get_name()})});
502  if (!result.successful) {
503  NodeT::undeclare_parameter(parameter->get_name());
504  throw exceptions::ComponentParameterException(result.reason);
505  }
506  }
507  } catch (const std::exception& ex) {
508  this->parameter_map_.remove_parameter(parameter->get_name());
509  throw exceptions::ComponentParameterException("Failed to add parameter: " + std::string(ex.what()));
510  }
511  } else {
512  RCLCPP_WARN_STREAM(this->get_logger(), "Parameter '" << parameter->get_name() << "' already exists.");
513  }
514 }

549 template<class NodeT>
550 inline rcl_interfaces::msg::SetParametersResult
551 ComponentInterface<NodeT>::on_set_parameters_callback(const std::vector<rclcpp::Parameter>& parameters) {
552  rcl_interfaces::msg::SetParametersResult result;
553  result.successful = true;
554  for (const auto& ros_parameter: parameters) {
555  try {
556  if (ros_parameter.get_name().substr(0, 27) == "qos_overrides./tf.publisher") {
557  continue;
558  }
559  // get the associated parameter interface by name
560  auto parameter = parameter_map_.get_parameter(ros_parameter.get_name());
561 
562  // convert the ROS parameter into a ParameterInterface without modifying the original
563  auto new_parameter = modulo_core::translators::read_parameter_const(ros_parameter, parameter);
564  if (!this->validate_parameter(new_parameter)) {
565  result.successful = false;
566  result.reason += "Validation of parameter '" + ros_parameter.get_name() + "' returned false!";
567  } else if (!new_parameter->is_empty()) {
568  // update the value of the parameter in the map
569  modulo_core::translators::copy_parameter_value(new_parameter, parameter);
570  }
571  } catch (const std::exception& ex) {
572  result.successful = false;
573  result.reason += ex.what();
574  }
575  }
576  this->set_parameter_callback_called_ = true;
577  return result;
578 }

612 template<class NodeT>
613 inline bool ComponentInterface<NodeT>::get_predicate(const std::string& predicate_name) {
614  auto predicate_iterator = this->predicates_.find(predicate_name);
615  // if there is no predicate with that name simply return false with an error message
616  if (predicate_iterator == this->predicates_.end()) {
617  RCLCPP_ERROR_STREAM_THROTTLE(this->get_logger(), *this->get_clock(), 1000,
618  "Failed to get predicate '" << predicate_name
619  << "': Predicate does not exists, returning false.");
620  return false;
621  }
622  // try to get the value from the variant as a bool
623  auto* ptr_value = std::get_if<bool>(&predicate_iterator->second);
624  if (ptr_value) {
625  return *ptr_value;
626  }
627  // if previous check failed, it means the variant is actually a callback function
628  auto callback_function = std::get<std::function<bool(void)>>(predicate_iterator->second);
629  bool value = false;
630  try {
631  value = (callback_function)();
632  } catch (const std::exception& ex) {
633  RCLCPP_ERROR_STREAM_THROTTLE(this->get_logger(), *this->get_clock(), 1000,
634  "Failed to evaluate callback of predicate'" << predicate_name << "', returning false:"
635  << ex.what());
636  }
637  return value;
638 }

713 template<class NodeT>
714 template<typename DataT>
715 inline void ComponentInterface<NodeT>::add_input(
716  const std::string& signal_name, const std::shared_ptr<DataT>& data, const std::string& default_topic,
717  bool fixed_topic
718 ) {
719  using namespace modulo_core::communication;
720  try {
721  std::string parsed_signal_name = this->validate_input_signal_name(signal_name);
722  std::string topic_name = default_topic.empty() ? "~/" + parsed_signal_name : default_topic;
723  auto parameter_name = parsed_signal_name + "_topic";
724  if (NodeT::has_parameter(parameter_name) && this->get_parameter(parameter_name)->is_empty()) {
725  this->set_parameter_value<std::string>(parameter_name, topic_name);
726  } else {
727  this->add_parameter(
728  parameter_name, topic_name, "Input topic name of signal '" + parsed_signal_name + "'", fixed_topic
729  );
730  }
731  topic_name = this->get_parameter_value<std::string>(parsed_signal_name + "_topic");
732  RCLCPP_DEBUG_STREAM(this->get_logger(),
733  "Adding input '" << signal_name << "' with topic name '" << topic_name << "'.");
734  auto message_pair = make_shared_message_pair(data, this->get_clock());
735  std::shared_ptr<SubscriptionInterface> subscription_interface;
736  switch (message_pair->get_type()) {
737  case MessageType::BOOL: {
738  auto subscription_handler = std::make_shared<SubscriptionHandler<std_msgs::msg::Bool>>(message_pair);
739  auto subscription = NodeT::template create_subscription<std_msgs::msg::Bool>(
740  topic_name, this->qos_, subscription_handler->get_callback());
741  subscription_interface = subscription_handler->create_subscription_interface(subscription);
742  break;
743  }
744  case MessageType::FLOAT64: {
745  auto subscription_handler = std::make_shared<SubscriptionHandler<std_msgs::msg::Float64>>(message_pair);
746  auto subscription = NodeT::template create_subscription<std_msgs::msg::Float64>(
747  topic_name, this->qos_, subscription_handler->get_callback());
748  subscription_interface = subscription_handler->create_subscription_interface(subscription);
749  break;
750  }
751  case MessageType::FLOAT64_MULTI_ARRAY: {
752  auto subscription_handler =
753  std::make_shared<SubscriptionHandler<std_msgs::msg::Float64MultiArray>>(message_pair);
754  auto subscription = NodeT::template create_subscription<std_msgs::msg::Float64MultiArray>(
755  topic_name, this->qos_, subscription_handler->get_callback());
756  subscription_interface = subscription_handler->create_subscription_interface(subscription);
757  break;
758  }
759  case MessageType::INT32: {
760  auto subscription_handler = std::make_shared<SubscriptionHandler<std_msgs::msg::Int32>>(message_pair);
761  auto subscription = NodeT::template create_subscription<std_msgs::msg::Int32>(
762  topic_name, this->qos_, subscription_handler->get_callback());
763  subscription_interface = subscription_handler->create_subscription_interface(subscription);
764  break;
765  }
766  case MessageType::STRING: {
767  auto subscription_handler = std::make_shared<SubscriptionHandler<std_msgs::msg::String>>(message_pair);
768  auto subscription = NodeT::template create_subscription<std_msgs::msg::String>(
769  topic_name, this->qos_, subscription_handler->get_callback());
770  subscription_interface = subscription_handler->create_subscription_interface(subscription);
771  break;
772  }
773  case MessageType::ENCODED_STATE: {
774  auto subscription_handler = std::make_shared<SubscriptionHandler<modulo_core::EncodedState>>(message_pair);
775  auto subscription = NodeT::template create_subscription<modulo_core::EncodedState>(
776  topic_name, this->qos_, subscription_handler->get_callback());
777  subscription_interface = subscription_handler->create_subscription_interface(subscription);
778  break;
779  }
780  }
781  this->inputs_.insert_or_assign(parsed_signal_name, subscription_interface);
782  } catch (const std::exception& ex) {
783  RCLCPP_ERROR_STREAM(this->get_logger(), "Failed to add input '" << signal_name << "': " << ex.what());
784  }
785 }
