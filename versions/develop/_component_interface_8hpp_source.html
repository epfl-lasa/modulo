534template<class NodeT>
535inline void ComponentInterface<NodeT>::add_parameter(
536 const std::shared_ptr<state_representation::ParameterInterface>& parameter, const std::string& description,
537 bool read_only
538) {
539 this->set_parameter_callback_called_ = false;
540 rclcpp::Parameter ros_param;
541 try {
542 ros_param = modulo_core::translators::write_parameter(parameter);
543 } catch (const modulo_core::exceptions::ParameterTranslationException& ex) {
544 throw exceptions::ComponentParameterException("Failed to add parameter: " + std::string(ex.what()));
545 }
546 if (!NodeT::has_parameter(parameter->get_name())) {
547 RCLCPP_DEBUG_STREAM(this->get_logger(), "Adding parameter '" << parameter->get_name() << "'.");
548 this->parameter_map_.set_parameter(parameter);
549 try {
550 rcl_interfaces::msg::ParameterDescriptor descriptor;
551 descriptor.description = description;
552 descriptor.read_only = read_only;
553 if (parameter->is_empty()) {
554 descriptor.dynamic_typing = true;
555 descriptor.type = modulo_core::translators::get_ros_parameter_type(parameter->get_parameter_type());
556 NodeT::declare_parameter(parameter->get_name(), rclcpp::ParameterValue{}, descriptor);
557 } else {
558 NodeT::declare_parameter(parameter->get_name(), ros_param.get_parameter_value(), descriptor);
559 }
560 if (!this->set_parameter_callback_called_) {
561 auto result = this->on_set_parameters_callback({NodeT::get_parameters({parameter->get_name()})});
562 if (!result.successful) {
563 NodeT::undeclare_parameter(parameter->get_name());
564 throw exceptions::ComponentParameterException(result.reason);
565 }
566 }
567 } catch (const std::exception& ex) {
568 this->parameter_map_.remove_parameter(parameter->get_name());
569 throw exceptions::ComponentParameterException("Failed to add parameter: " + std::string(ex.what()));
570 }
571 } else {
572 RCLCPP_WARN_STREAM(this->get_logger(), "Parameter '" << parameter->get_name() << "' already exists.");
573 }
574}

623template<class NodeT>
624inline rcl_interfaces::msg::SetParametersResult
625ComponentInterface<NodeT>::on_set_parameters_callback(const std::vector<rclcpp::Parameter>& parameters) {
626 rcl_interfaces::msg::SetParametersResult result;
627 result.successful = true;
628 for (const auto& ros_parameter: parameters) {
629 try {
630 if (ros_parameter.get_name().substr(0, 27) == "qos_overrides./tf.publisher") {
631 continue;
632 }
633 // get the associated parameter interface by name
634 auto parameter = parameter_map_.get_parameter(ros_parameter.get_name());
635
636 // convert the ROS parameter into a ParameterInterface without modifying the original
637 auto new_parameter = modulo_core::translators::read_parameter_const(ros_parameter, parameter);
638 if (!this->validate_parameter(new_parameter)) {
639 result.successful = false;
640 result.reason += "Validation of parameter '" + ros_parameter.get_name() + "' returned false!";
641 } else if (!new_parameter->is_empty()) {
642 // update the value of the parameter in the map
643 modulo_core::translators::copy_parameter_value(new_parameter, parameter);
644 }
645 } catch (const std::exception& ex) {
646 result.successful = false;
647 result.reason += ex.what();
648 }
649 }
650 this->set_parameter_callback_called_ = true;
651 return result;
652}

666template<class NodeT>
667inline void ComponentInterface<NodeT>::add_variant_predicate(
668 const std::string& name, const utilities::PredicateVariant& predicate
669) {
670 if (name.empty()) {
671 RCLCPP_ERROR(this->get_logger(), "Failed to add predicate: Provide a non empty string as a name.");
672 return;
673 }
674 if (this->predicates_.find(name) != this->predicates_.end()) {
675 RCLCPP_WARN_STREAM(this->get_logger(), "Predicate '" << name << "' already exists, overwriting.");
676 } else {
677 RCLCPP_DEBUG_STREAM(this->get_logger(), "Adding predicate '" << name << "'.");
678 auto publisher = rclcpp::create_publisher<std_msgs::msg::Bool>(
679 *this, utilities::generate_predicate_topic(this->get_name(), name), this->qos_
680 );
681 this->predicate_publishers_.insert_or_assign(name, publisher);
682 }
683 this->predicates_.insert_or_assign(name, predicate);
684}

686template<class NodeT>
687inline bool ComponentInterface<NodeT>::get_predicate(const std::string& predicate_name) {
688 auto predicate_iterator = this->predicates_.find(predicate_name);
689 // if there is no predicate with that name simply return false with an error message
690 if (predicate_iterator == this->predicates_.end()) {
691 RCLCPP_ERROR_STREAM_THROTTLE(this->get_logger(), *this->get_clock(), 1000,
692 "Failed to get predicate '" << predicate_name
693 << "': Predicate does not exists, returning false.");
694 return false;
695 }
696 // try to get the value from the variant as a bool
697 auto* ptr_value = std::get_if<bool>(&predicate_iterator->second);
698 if (ptr_value) {
699 return *ptr_value;
700 }
701 // if previous check failed, it means the variant is actually a callback function
702 auto callback_function = std::get<std::function<bool(void)>>(predicate_iterator->second);
703 bool value = false;
704 try {
705 value = (callback_function)();
706 } catch (const std::exception& ex) {
707 RCLCPP_ERROR_STREAM_THROTTLE(this->get_logger(), *this->get_clock(), 1000,
708 "Failed to evaluate callback of predicate '" << predicate_name
709 << "', returning false: " << ex.what());
710 }
711 return value;
712}

714template<class NodeT>
715inline void ComponentInterface<NodeT>::add_trigger(const std::string& trigger_name) {
716 if (trigger_name.empty()) {
717 RCLCPP_ERROR(this->get_logger(), "Failed to add trigger: Provide a non empty string as a name.");
718 return;
719 }
720 if (this->triggers_.find(trigger_name) != this->triggers_.end()
721 || this->predicates_.find(trigger_name) != this->predicates_.end()) {
722 RCLCPP_ERROR_STREAM(this->get_logger(), "Failed to add trigger: there is already a trigger or "
723 "predicate with name '" << trigger_name << "'.");
724 return;
725 }
726 this->triggers_.insert_or_assign(trigger_name, false);
727 this->add_predicate(
728 trigger_name, [this, &trigger_name] {
729 auto value = this->triggers_.at(trigger_name);
730 this->triggers_.at(trigger_name) = false;
731 return value;
732 }
733 );
734}

787template<class NodeT>
788template<typename DataT>
789inline void ComponentInterface<NodeT>::add_input(
790 const std::string& signal_name, const std::shared_ptr<DataT>& data, const std::string& default_topic,
791 bool fixed_topic
792) {
793 using namespace modulo_core::communication;
794 try {
795 std::string parsed_signal_name = this->validate_input_signal_name(signal_name);
796 if (data == nullptr) {
797 throw modulo_core::exceptions::NullPointerException(
798 "Invalid data pointer for input '" + parsed_signal_name + "'.");
799 }
800 std::string topic_name = default_topic.empty() ? "~/" + parsed_signal_name : default_topic;
801 auto parameter_name = parsed_signal_name + "_topic";
802 if (NodeT::has_parameter(parameter_name) && this->get_parameter(parameter_name)->is_empty()) {
803 this->set_parameter_value<std::string>(parameter_name, topic_name);
804 } else {
805 this->add_parameter(
806 parameter_name, topic_name, "Signal topic name of input '" + parsed_signal_name + "'", fixed_topic
807 );
808 }
809 topic_name = this->get_parameter_value<std::string>(parameter_name);
810 RCLCPP_DEBUG_STREAM(this->get_logger(),
811 "Adding input '" << parsed_signal_name << "' with topic name '" << topic_name << "'.");
812 auto message_pair = make_shared_message_pair(data, this->get_clock());
813 std::shared_ptr<SubscriptionInterface> subscription_interface;
814 switch (message_pair->get_type()) {
815 case MessageType::BOOL: {
816 auto subscription_handler = std::make_shared<SubscriptionHandler<std_msgs::msg::Bool>>(message_pair);
817 auto subscription = NodeT::template create_subscription<std_msgs::msg::Bool>(
818 topic_name, this->qos_, subscription_handler->get_callback());
819 subscription_interface = subscription_handler->create_subscription_interface(subscription);
820 break;
821 }
822 case MessageType::FLOAT64: {
823 auto subscription_handler = std::make_shared<SubscriptionHandler<std_msgs::msg::Float64>>(message_pair);
824 auto subscription = NodeT::template create_subscription<std_msgs::msg::Float64>(
825 topic_name, this->qos_, subscription_handler->get_callback());
826 subscription_interface = subscription_handler->create_subscription_interface(subscription);
827 break;
828 }
829 case MessageType::FLOAT64_MULTI_ARRAY: {
830 auto subscription_handler =
831 std::make_shared<SubscriptionHandler<std_msgs::msg::Float64MultiArray>>(message_pair);
832 auto subscription = NodeT::template create_subscription<std_msgs::msg::Float64MultiArray>(
833 topic_name, this->qos_, subscription_handler->get_callback());
834 subscription_interface = subscription_handler->create_subscription_interface(subscription);
835 break;
836 }
837 case MessageType::INT32: {
838 auto subscription_handler = std::make_shared<SubscriptionHandler<std_msgs::msg::Int32>>(message_pair);
839 auto subscription = NodeT::template create_subscription<std_msgs::msg::Int32>(
840 topic_name, this->qos_, subscription_handler->get_callback());
841 subscription_interface = subscription_handler->create_subscription_interface(subscription);
842 break;
843 }
844 case MessageType::STRING: {
845 auto subscription_handler = std::make_shared<SubscriptionHandler<std_msgs::msg::String>>(message_pair);
846 auto subscription = NodeT::template create_subscription<std_msgs::msg::String>(
847 topic_name, this->qos_, subscription_handler->get_callback());
848 subscription_interface = subscription_handler->create_subscription_interface(subscription);
849 break;
850 }
851 case MessageType::ENCODED_STATE: {
852 auto subscription_handler = std::make_shared<SubscriptionHandler<modulo_core::EncodedState>>(message_pair);
853 auto subscription = NodeT::template create_subscription<modulo_core::EncodedState>(
854 topic_name, this->qos_, subscription_handler->get_callback());
855 subscription_interface = subscription_handler->create_subscription_interface(subscription);
856 break;
857 }
858 }
859 this->inputs_.insert_or_assign(parsed_signal_name, subscription_interface);
860 } catch (const std::exception& ex) {
861 RCLCPP_ERROR_STREAM(this->get_logger(), "Failed to add input '" << signal_name << "': " << ex.what());
862 }
863}

1037template<class NodeT>
1038template<typename T>
1039inline void ComponentInterface<NodeT>::publish_transforms(
1040 const std::vector<state_representation::CartesianPose>& transforms, const std::shared_ptr<T>& tf_broadcaster,
1041 bool is_static
1042) {
1043 std::string modifier = is_static ? "static " : "";
1044 if (tf_broadcaster == nullptr) {
1045 RCLCPP_ERROR_STREAM_THROTTLE(
1046 this->get_logger(), *this->get_clock(), 1000,
1047 "Failed to send " << modifier << "transform: No " << modifier << "TF broadcaster configured.");
1048 return;
1049 }
1050 try {
1051 std::vector<geometry_msgs::msg::TransformStamped> transform_messages;
1052 transform_messages.reserve(transforms.size());
1053 for (const auto& tf: transforms) {
1054 geometry_msgs::msg::TransformStamped transform_message;
1055 modulo_core::translators::write_message(transform_message, tf, this->get_clock()->now());
1056 transform_messages.emplace_back(transform_message);
1057 }
1058 tf_broadcaster->sendTransform(transform_messages);
1059 } catch (const std::exception& ex) {
1060 RCLCPP_ERROR_STREAM_THROTTLE(this->get_logger(), *this->get_clock(), 1000,
1061 "Failed to send " << modifier << "transform: " << ex.what());
1062 }

1067 const std::string& frame, const std::string& reference_frame, const tf2::TimePoint& time_point,
1068 const tf2::Duration& duration
1069) {
1070 if (this->tf_buffer_ == nullptr || this->tf_listener_ == nullptr) {
1071 throw exceptions::LookupTransformException("Failed to lookup transform: To TF buffer / listener configured.");
1072 }

1074 state_representation::CartesianPose result(frame, reference_frame);
