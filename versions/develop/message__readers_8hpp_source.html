179template<typename T>
180inline std::shared_ptr<T> safe_dynamic_pointer_cast(std::shared_ptr<state_representation::State>& state) {
181 auto derived_state_ptr = std::dynamic_pointer_cast<T>(state);
182 if (derived_state_ptr == nullptr) {
183 throw modulo_core::exceptions::MessageTranslationException(
184 "Dynamic casting of state " + state->get_name() + " failed.");
185 }
186 return derived_state_ptr;

224template<typename StateT, typename NewStateT>
225inline void safe_spatial_state_conversion(
226 std::shared_ptr<state_representation::State>& state, std::shared_ptr<state_representation::State>& new_state,
227 std::function<void(StateT&, const NewStateT&)> conversion_callback = {}
228) {
229 auto derived_state_ptr = safe_dynamic_pointer_cast<StateT>(state);
230 auto derived_new_state_ptr = safe_dynamic_pointer_cast<NewStateT>(new_state);
231 StateT tmp_new_state;
232 tmp_new_state.set_name(derived_new_state_ptr->get_name());
233 tmp_new_state.set_reference_frame(derived_new_state_ptr->get_reference_frame());
234 if (!derived_new_state_ptr->is_empty()) {
235 tmp_new_state.set_empty(derived_new_state_ptr->is_empty());
236 if (conversion_callback) {
237 conversion_callback(tmp_new_state, *derived_new_state_ptr);
238 }
239 }
240 *derived_state_ptr = tmp_new_state;
241}

277inline void read_message(std::shared_ptr<state_representation::State>& state, const EncodedState& message) {
278 using namespace state_representation;
279 std::string tmp(message.data.begin(), message.data.end());
280 std::shared_ptr<State> new_state;
281 try {
282 new_state = clproto::decode<std::shared_ptr<State>>(tmp);
283 } catch (const std::exception& ex) {
284 throw exceptions::MessageTranslationException(ex.what());
285 }
286 try {
287 switch (state->get_type()) {
288 case StateType::STATE: {
289 if (new_state->get_type() == StateType::STATE) {
290 *state = *new_state;
291 } else {
292 *state = State(StateType::STATE, new_state->get_name(), new_state->is_empty());
293 }
294 break;
295 }
296 case StateType::SPATIAL_STATE: {
297 std::set<StateType> spatial_state_types = {
298 StateType::SPATIAL_STATE, StateType::CARTESIAN_STATE, StateType::CARTESIAN_POSE, StateType::CARTESIAN_TWIST,
299 StateType::CARTESIAN_ACCELERATION, StateType::CARTESIAN_WRENCH
300 };
301 if (spatial_state_types.find(new_state->get_type()) != spatial_state_types.cend()) {
302 safe_spatial_state_conversion<SpatialState, SpatialState>(state, new_state);
303 } else {
304 safe_dynamic_cast<SpatialState>(state, new_state);
305 }
306 break;
307 }
308 case StateType::CARTESIAN_STATE: {
309 if (new_state->get_type() == StateType::CARTESIAN_POSE) {
310 safe_spatial_state_conversion<CartesianState, CartesianPose>(
311 state, new_state, [](CartesianState& a, const CartesianPose& b) {
312 a.set_pose(b.get_pose());
313 });
314 } else if (new_state->get_type() == StateType::CARTESIAN_TWIST) {
315 safe_spatial_state_conversion<CartesianState, CartesianTwist>(
316 state, new_state, [](CartesianState& a, const CartesianTwist& b) {
317 a.set_twist(b.get_twist());
318 });
319 } else if (new_state->get_type() == StateType::CARTESIAN_ACCELERATION) {
320 safe_spatial_state_conversion<CartesianState, CartesianAcceleration>(
321 state, new_state, [](CartesianState& a, const CartesianAcceleration& b) {
322 a.set_acceleration(b.get_acceleration());
323 });
324 } else if (new_state->get_type() == StateType::CARTESIAN_WRENCH) {
325 safe_spatial_state_conversion<CartesianState, CartesianWrench>(
326 state, new_state, [](CartesianState& a, const CartesianWrench& b) {
327 a.set_wrench(b.get_wrench());
328 });
329 } else {
330 safe_dynamic_cast<CartesianState>(state, new_state);
331 }
332 break;
333 }

335 if (new_state->get_type() == StateType::CARTESIAN_STATE) {

454 switch (param_ptr->get_parameter_type()) {
