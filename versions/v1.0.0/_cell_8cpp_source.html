339 rclcpp_lifecycle::node_interfaces::LifecycleNodeInterface::CallbackReturn Cell::on_configure(const rclcpp_lifecycle::State&) {
340  RCUTILS_LOG_INFO_NAMED(get_name(), "on_configure() is called.");
341  this->configured_ = true;
342  // call the proxy on_configure function
343  if (!this->on_configure()) {
344  RCLCPP_ERROR(get_logger(), "Configuration failed");
345  this->reset();
346  return rclcpp_lifecycle::node_interfaces::LifecycleNodeInterface::CallbackReturn::FAILURE;
347  }
348  // add the run periodic call
349  this->timers_.push_back(this->create_wall_timer(this->period_, [this] { this->run(); }));
350  // add default transform broadcaster and transform listener
351  this->add_transform_broadcaster(this->period_, true);
352  this->add_transform_listener(10 * this->period_);
353  return rclcpp_lifecycle::node_interfaces::LifecycleNodeInterface::CallbackReturn::SUCCESS;

466 void Cell::update_parameters() {
467  using namespace state_representation;
468  using namespace state_representation::exceptions;
469  try {
470  for (auto& [key, param] : this->parameters_) {
471  switch (param->get_type()) {
472  case StateType::PARAMETER_DOUBLE: {
473  double value = this->get_parameter(param->get_name()).as_double();
474  std::static_pointer_cast<Parameter<double>>(param)->set_value(value);
475  break;
476  }
477 
478  case StateType::PARAMETER_DOUBLE_ARRAY: {
479  std::vector<double> value = this->get_parameter(param->get_name()).as_double_array();
480  std::static_pointer_cast<Parameter<std::vector<double>>>(param)->set_value(value);
481  break;
482  }
483 
484  case StateType::PARAMETER_BOOL: {
485  bool value = this->get_parameter(param->get_name()).as_bool();
486  std::static_pointer_cast<Parameter<bool>>(param)->set_value(value);
487  break;
488  }
489 
490  case StateType::PARAMETER_BOOL_ARRAY: {
491  std::vector<bool> value = this->get_parameter(param->get_name()).as_bool_array();
492  std::static_pointer_cast<Parameter<std::vector<bool>>>(param)->set_value(value);
493  break;
494  }
495 
496  case StateType::PARAMETER_STRING: {
497  std::string value = this->get_parameter(param->get_name()).as_string();
498  std::static_pointer_cast<Parameter<std::string>>(param)->set_value(value);
499  break;
500  }
501 
502  case StateType::PARAMETER_STRING_ARRAY: {
503  std::vector<std::string> value = this->get_parameter(param->get_name()).as_string_array();
504  std::static_pointer_cast<Parameter<std::vector<std::string>>>(param)->set_value(value);
505  break;
506  }
507 
508  case StateType::PARAMETER_CARTESIANSTATE: {
509  std::vector<double> value = this->get_parameter(param->get_name()).as_double_array();
510  std::static_pointer_cast<Parameter<CartesianState>>(param)->get_value().set_data(value);
511  break;
512  }
513 
514  case StateType::PARAMETER_CARTESIANPOSE: {
515  std::vector<double> value = this->get_parameter(param->get_name()).as_double_array();
516  std::static_pointer_cast<Parameter<CartesianPose>>(param)->get_value().CartesianPose::set_data(value);
517  break;
518  }
519 
520  case StateType::PARAMETER_JOINTSTATE: {
521  std::vector<double> value = this->get_parameter(param->get_name()).as_double_array();
522  std::static_pointer_cast<Parameter<JointState>>(param)->get_value().set_data(value);
523  break;
524  }
525 
526  case StateType::PARAMETER_JOINTPOSITIONS: {
527  std::vector<double> value = this->get_parameter(param->get_name()).as_double_array();
528  std::static_pointer_cast<Parameter<JointPositions>>(param)->get_value().JointPositions::set_data(value);
529  break;
530  }
531 
532  case StateType::PARAMETER_ELLIPSOID: {
533  std::vector<double> value = this->get_parameter(param->get_name()).as_double_array();
534  std::static_pointer_cast<Parameter<Ellipsoid>>(param)->get_value().set_data(value);
535  break;
536  }
537 
538  case StateType::PARAMETER_MATRIX: {
539  std::vector<double> value = this->get_parameter(param->get_name()).as_double_array();
